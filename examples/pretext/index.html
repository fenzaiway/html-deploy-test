快速开始

import asyncio
from pretext import Client

async def main():
    async with Client() as client:
        # 发送 GET 请求
        resp = await client.get(
            "https://api.github.com/repos/python/cpython",
            headers={"Accept": "application/vnd.github+json"}
        )
        print(resp.status_code)    # 200
        print(resp.json()["stargazers_count"])

asyncio.run(main())

中间件示例

from pretext import Client, Request, Response

async def retry_middleware(req, next_fn):
    """请求失败自动重试 3 次"""
    for attempt in range(3):
        resp = await next_fn(req)
        if resp.status_code < 500:
            return resp
        print(f"Retry {attempt+1}/3...")
    return resp

async def main():
    async with Client(middleware=[retry_middleware]) as c:
        resp = await c.get("https://flaky-api.example.com/data")
        print(resp.text)
